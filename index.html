Grade Calculator

from pyscript import Element def calculate_grades(event): try: attendance = float(Element("attendance").element.value) * 0.1 recitation = float(Element("recitation").element.value) * 0.4 quiz = float(Element("quiz").element.value) * 0.3 requirements = float(Element("requirements").element.value) * 0.3 prelim_exam = float(Element("prelim_exam").element.value) * 0.6 performance_task = recitation + quiz + requirements overall_performance = performance_task * 0.3 prelim_grade = round(attendance + overall_performance + prelim_exam, 2) result_element = Element("result").element if float(Element("attendance").element.value) <= 60: result_element.className = "failed" result_text = f"Your Prelim Grade is: {prelim_grade:.2f} - Failed due to excessive absences." else: result_element.className = "passed" result_text = f"Your Prelim Grade is: {prelim_grade:.2f} - PASSED" Element("resultText").element.innerHTML = result_text result_element.style.display = "block" except ValueError: Element("resultText").element.innerHTML = "Invalid input. Please enter numbers only." Element("result").element.style.display = "block" Element("calculateButton").element.onclick = calculate_grades
